  0x01,
  0x00,                                /* bAssocTerminal */

  0x00,                                /* iTerminal */

  0x00,                                 /* iInterface */

  /* USB Speaker Standard AS Interface Descriptor - Audio Streaming Operational */

  AUDIO_INTERFACE_DESC_SIZE,            /* bLength */

  0x01,                                 /* bInterfaceNumber */
